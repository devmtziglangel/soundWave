package modelo.usuarios;

import java.util.ArrayList;
import java.util.Date;
import java.util.UUID;
import enums.TipoSuscripcion;
import modelo.contenido.Contenido;
import modelo.plataforma.Playlist;
import excepciones.usuario.EmailInvalidoException;
import excepciones.usuario.PasswordDebilException;

public abstract class Usuario {

    // ==========================================
    // 1. ATRIBUTOS (Protected para herencia)
    // ==========================================
    protected String id;
    protected String nombre;
    protected String email;
    protected String password;
    protected TipoSuscripcion suscripcion;
    
    protected ArrayList<Playlist> misPlaylists;
    protected ArrayList<Contenido> historial;
    protected Date fechaRegistro;

    // ==========================================
    // 2. CONSTRUCTOR
    // ==========================================
    public Usuario(String nombre, String email, String password, TipoSuscripcion suscripcion) 
            throws EmailInvalidoException, PasswordDebilException {
        
        this.id = UUID.randomUUID().toString(); // ID inmutable generado automáticamente
        this.nombre = nombre;
        
        // Asignamos directamente para poder validar
        this.email = email;
        validarEmail(); // Lanza excepción si falla
        
        this.password = password;
        validarPassword(); // Lanza excepción si falla
        
        this.suscripcion = suscripcion;
        
        // Inicialización de colecciones para evitar NullPointerException
        this.misPlaylists = new ArrayList<>();
        this.historial = new ArrayList<>();
        this.fechaRegistro = new Date();
    }

    // ==========================================
    // 3. GETTERS Y SETTERS (Intercalados)
    // ==========================================
    
    // ID (Solo lectura, no tiene setter)
    public String getId() {
        return id;
    }

    // Nombre
    public String getNombre() {
        return nombre;
    }

    public void setNombre(String nombre) {
        this.nombre = nombre;
    }

    // Email (Setter con validación)
    public String getEmail() {
        return email;
    }

    public void setEmail(String nuevoEmail) throws EmailInvalidoException {
        String emailAnterior = this.email;
        this.email = nuevoEmail;
        try {
            validarEmail();
        } catch (EmailInvalidoException e) {
            this.email = emailAnterior; // Restaurar valor seguro si falla
            throw e; 
        }
    }

    // Password (Setter con validación de seguridad)
    // Nota: A veces se omite getPassword por seguridad, pero lo dejamos por si se necesita
    public String getPassword() {
        return password;
    }

    public void setPassword(String nuevaPassword) throws PasswordDebilException {
        String passwordAnterior = this.password;
        this.password = nuevaPassword;
        try {
            validarPassword();
        } catch (PasswordDebilException e) {
            this.password = passwordAnterior; // Restaurar valor seguro
            throw e;
        }
    }

    // Suscripción
    public TipoSuscripcion getSuscripcion() {
        return suscripcion;
    }

    public void setSuscripcion(TipoSuscripcion suscripcion) {
        this.suscripcion = suscripcion;
    }

    // Listas y Fecha (Solo lectura de la referencia, manipulación vía métodos)
    public ArrayList<Playlist> getMisPlaylists() {
        return misPlaylists;
    }

    public ArrayList<Contenido> getHistorial() {
        return historial;
    }

    public Date getFechaRegistro() {
        return fechaRegistro;
    }

    // ==========================================
    // 4. MÉTODOS DE NEGOCIO Y VALIDACIÓN
    // ==========================================

    /**
     * Método abstracto que obliga a las clases hijas (Premium/Gratuito)
     * a definir su propia forma de reproducir.
     */
    public abstract void reproducir(Contenido contenido);

    public void crearPlaylist(String nombre) {
        // 'this' pasa la referencia del usuario actual como creador
        Playlist nuevaPlaylist = new Playlist(nombre, this);
        this.misPlaylists.add(nuevaPlaylist);
        System.out.println("Playlist '" + nombre + "' creada exitosamente.");
    }

    public void seguirPlaylist(Playlist playlist) {
        if (!this.misPlaylists.contains(playlist)) {
            this.misPlaylists.add(playlist);
            System.out.println(this.nombre + " ahora sigue la playlist: " + playlist.getNombre());
        }
    }

    public void darLike(Contenido contenido) {
        contenido.agregarLike();
        System.out.println("A " + this.nombre + " le gusta: " + contenido.getTitulo());
    }

    public void agregarAlHistorial(Contenido contenido) {
        this.historial.add(contenido);
    }

    // --- Métodos de Validación ---

    public void validarEmail() throws EmailInvalidoException {
        if (this.email == null || !this.email.contains("@") || !this.email.contains(".")) {
            throw new EmailInvalidoException("El email '" + this.email + "' no tiene un formato válido (debe contener @ y .).");
        }
    }

    public void validarPassword() throws PasswordDebilException {
        if (this.password == null || this.password.length() < 8) {
            throw new PasswordDebilException("La contraseña es muy débil (mínimo 8 caracteres).");
        }
    }
}
